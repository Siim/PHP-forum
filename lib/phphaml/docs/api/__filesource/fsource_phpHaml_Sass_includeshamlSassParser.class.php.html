  166.     public function render($file = null, $type = null)
    167. 

  167.     {
    168. 

  168.         if (!is_null($file))
    169. 

  169.             $this->setFile($file);
    170. 

  170.         if (!is_null($type))
    171. 

  171.             $this->setRenderer($type);
    172. 

  172.         // BEGIN: Flat to tree structure
    173. 

  173.         $lines = explode(self::TOKEN_LINE, $this->getSource());
    174. 

  174.         $last = array(-1 => null);
    175. 

  175.         $attributes = array();
    176. 

  176.         $attributeLevel = 0;
    177. 

  177.         $namespace = null;
    178. 

  178.         $namespaceLevel = 0;
    179. 

  179.         foreach ($lines as $line)
    180. 

  180.         {
    181. 

  181.             $stop = true;
    182. 

  182.             $cleaned = $this->cleanLine($line);
    183. 

  183.             $level = $this->countLevel($line);
    184. 

  184.             // Check for constant definition
    185. 

  185.             if (preg_match('/^'.preg_quote(self::TOKEN_CONSTANT).'(.+?)[ ]?'.preg_quote(self::TOKEN_CONSTANT_VALUE).'[ ]?["]?(.+)["]?/', $cleaned, $matches))
    186. 

  186.                 $this->setConstant($matches[1], $matches[2]); else
    187. 

  187.             // Check for dynamic attribute definition
    188. 

  188.             if (preg_match('/^'.preg_quote(self::TOKEN_ATTRIBUTE).'(.+?)[ ]?'.preg_quote(self::TOKEN_ATTRIBUTE_CALCULATE ).'[ ]?(.+)/', $cleaned, $matches))
    189. 

  189.             {
    190. 

  190.                 if ($namespaceLevel+1 != $level)
    191. 

  191.                     $namespace = null;
    192. 

  192.                 $attributes[empty($namespace) ? $matches[1] : "$namespace-{$matches[1]}"] = $this->calculate($matches[2]);
    193. 

  193.                 $attributeLevel = $level - (empty($namespace) ? 1 : 2);
    194. 

  194.             } else
    195. 

  195.             // Check for attribute definition
    196. 

  196.             if (preg_match('/^'.preg_quote(self::TOKEN_ATTRIBUTE).'(.+?) (.+)/', $cleaned, $matches))
    197. 

  197.             {
    198. 

  198.                 if ($namespaceLevel+1 != $level)
    199. 

  199.                     $namespace = null;
    200. 

  200.                 $attributes[empty($namespace) ? $matches[1] : "$namespace-{$matches[1]}"] = $matches[2];
    201. 

  201.                 $attributeLevel = $level - (empty($namespace) ? 1 : 2);
    202. 

  202.             } else
    203. 

  203.             // Check for attribute namespace
    204. 

  204.             if (preg_match('/^'.preg_quote(self::TOKEN_ATTRIBUTE_NAMESPACE).'(.+)/', $cleaned, $matches))
    205. 

  205.             {
    206. 

  206.                 $namespace = $matches[1];
    207. 

  207.                 $namespaceLevel = $level;
    208. 

  208.             } else
    209. 

  209.             // Check for comment
    210. 

  210.             if (preg_match('|^'.preg_quote(self::TOKEN_COMMENT).'|', $cleaned))
    211. 

  211.                 $stop = true;
    212. 

  212.             // Nothing special
    213. 

  213.             else
    214. 

  214.             {
    215. 

  215.                 $stop = false;
    216. 

  216.                 $namespace = null;
    217. 

  217.             }
    218. 

  218.             // Remove blank lines
    219. 

  219.             if (empty($cleaned) || $stop)
    220. 

  220.                 continue;
    221. 

  221.             // Assign attributes
    222. 

  222.             if (!empty($attributes) && $last[$attributeLevel] instanceof SassElement)
    223. 

  223.             {
    224. 

  224.                 foreach ($attributes as $name => $value)
    225. 

  225.                     $last[$attributeLevel]->setAttribute($name, $value);
    226. 

  226.                 $attributeLevel = 0;
    227. 

  227.                 $attributes = array();
    228. 

  228.             }
    229. 

  229.             // Get parent element
    230. 

  230.             $parent = $last[$level - 1];
    231. 

  231.             // Create new element
    232. 

  232.             $element = new SassElement($parent, $cleaned);
    233. 

  233.             // Mark as parent
    234. 

  234.             $last[$level] = $element;
    235. 

  235.             if ($level == 0)
    236. 

  236.                 $this->addElement($element);
    237. 

  237.         }
    238. 

  238.         // Assign attributes
    239. 

  239.         if (!empty($attributes) && $last[$attributeLevel] instanceof SassElement)
    240. 

  240.         {
    241. 

  241.             foreach ($attributes as $name => $value)
    242. 

  242.                 $last[$attributeLevel]->setAttribute($name, $value);
    243. 

  243.             $attributeLevel = 0;
    244. 

  244.             $attributes = array();
    245. 

  245.         }
    246. 

  246.         // END: Flat to tree structure
    247. 

  247.         // Render to CSS
    248. 

  248.         return SassRenderer::getInstance($this->getElements(), $this->getRenderer())->render();
    249. 

  249.     }
