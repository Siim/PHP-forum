  234.     public function render($file = null, $type = null)
    235. 

  235.     {
    236. 

  236.         if (!is_null($file))
    237. 

  237.             $this->setFile($file);

  238.         if (!is_null($type))
    239. 

  239.             $this->setRenderer($type);
    240. 

  240.         $cache = new CommonCache($this->getTmp(), 'css', $this->getSource());
    241. 

  241.         if (!$cache->isCached())
    242. 

  242.         {
    243. 

  243.             // BEGIN: Flat to tree structure
    244. 

  244.             $lines = explode(self::TOKEN_LINE, $this->getSource());
    245. 

  245.             $last = array(-1 => null);
    246. 

  246.             $attributes = array();
    247. 

  247.             $attributeLevel = 0;
    248. 

  248.             $namespace = null;
    249. 

  249.             $namespaceLevel = 0;
    250. 

  250.             foreach ($lines as $line)
    251. 

  251.             {
    252. 

  252.                 $stop = true;
    253. 

  253.                 $cleaned = $this->cleanLine($line);
    254. 

  254.                 $level = $this->countLevel($line);
    255. 

  255.                 // Check for constant definition
    256. 

  256.                 if (preg_match('/^'.preg_quote(self::TOKEN_CONSTANT).'(.+?)[ ]?'.preg_quote(self::TOKEN_CONSTANT_VALUE).'[ ]?["]?(.+)["]?/', $cleaned, $matches))
    257. 

  257.                     $this->setConstant($matches[1], $matches[2]); else
    258. 

  258.                 // Check for dynamic attribute definition
    259. 

  259.                 if (preg_match('/^'.preg_quote(self::TOKEN_ATTRIBUTE).'(.+?)[ ]?'.preg_quote(self::TOKEN_ATTRIBUTE_CALCULATE ).'[ ]?(.+)/', $cleaned, $matches))
    260. 

  260.                 {
    261. 

  261.                     if ($namespaceLevel+1 != $level)
    262. 

  262.                         $namespace = null;
    263. 

  263.                     $attributes[empty($namespace) ? $matches[1] : "$namespace-{$matches[1]}"] = $this->calculate($matches[2]);
    264. 

  264.                     $attributeLevel = $level - (empty($namespace) ? 1 : 2);
    265. 

  265.                 } else
    266. 

  266.                 // Check for attribute definition
    267. 

  267.                 if (preg_match('/^'.preg_quote(self::TOKEN_ATTRIBUTE).'(.+?) (.+)/', $cleaned, $matches))
    268. 

  268.                 {
    269. 

  269.                     if ($namespaceLevel+1 != $level)
    270. 

  270.                         $namespace = null;
    271. 

  271.                     $attributes[empty($namespace) ? $matches[1] : "$namespace-{$matches[1]}"] = $matches[2];
    272. 

  272.                     $attributeLevel = $level - (empty($namespace) ? 1 : 2);
    273. 

  273.                 } else
    274. 

  274.                 // Check for attribute namespace
    275. 

  275.                 if (preg_match('/^'.preg_quote(self::TOKEN_ATTRIBUTE_NAMESPACE).'(.+)/', $cleaned, $matches))
    276. 

  276.                 {
    277. 

  277.                     $namespace = $matches[1];
    278. 

  278.                     $namespaceLevel = $level;
    279. 

  279.                 } else
    280. 

  280.                 // Check for comment
    281. 

  281.                 if (preg_match('|^'.preg_quote(self::TOKEN_COMMENT).'|', $cleaned))
    282. 

  282.                     $stop = true;
    283. 

  283.                 // Nothing special
    284. 

  284.                 else
    285. 

  285.                 {
    286. 

  286.                     $stop = false;
    287. 

  287.                     $namespace = null;
    288. 

  288.                 }
    289. 

  289.                 // Remove blank lines
    290. 

  290.                 if (empty($cleaned) || $stop)
    291. 

  291.                     continue;
    292. 

  292.                 // Assign attributes
    293. 

  293.                 if (!empty($attributes) && $last[$attributeLevel] instanceof SassElement)
    294. 

  294.                 {
    295. 

  295.                     foreach ($attributes as $name => $value)
    296. 

  296.                         $last[$attributeLevel]->setAttribute($name, $value);
    297. 

  297.                     $attributeLevel = 0;
    298. 

  298.                     $attributes = array();
    299. 

  299.                 }
    300. 

  300.                 // Get parent element
    301. 

  301.                 $parent = $last[$level - 1];
    302. 

  302.                 // Create new element
    303. 

  303.                 $element = new SassElement($parent, $cleaned);
    304. 

  304.                 // Mark as parent
    305. 

  305.                 $last[$level] = $element;
    306. 

  306.                 if ($level == 0)
    307. 

  307.                     $this->addElement($element);
    308. 

  308.             }
    309. 

  309.             // Assign attributes
    310. 

  310.             if (!empty($attributes) && $last[$attributeLevel] instanceof SassElement)
    311. 

  311.             {
    312. 

  312.                 foreach ($attributes as $name => $value)
    313. 

  313.                     $last[$attributeLevel]->setAttribute($name, $value);
    314. 

  314.                 $attributeLevel = 0;
    315. 

  315.                 $attributes = array();
    316. 

  316.             }
    317. 

  317.             // END: Flat to tree structure
    318. 

  318.             // Render to CSS
    319. 

  319.             return $cache->setCached(SassRenderer::getInstance($this->getElements(), $this->getRenderer())->render())->cacheIt()->getCached();
    320. 

  320.         }
    321. 

  321.         else
    322. 

  322.             return $cache->getCached();
    323. 

  323.     }

    459. 

  459.     public function calculate($expression)
    460. 

  460.     {
    461. 

  461.         // Replace constants with values
    462. 

  462.         foreach ($this->getConstants() as $name => $value)
    463. 

  463.             $expression = str_ireplace(self::TOKEN_CONSTANT.$name, $value, $expression);
    464. 

  464.         // Replace colors names with RGB codes
    465. 

  465.         foreach (self::$colors as $name => $rgb)
    466. 

  466.             $expression = str_ireplace($name, 'rgb('.implode(', ', $rgb).')', $expression);
    467. 

  467.         // Replace colors representation in HEX to rgb()
    468. 

  468.         $expression = preg_replace_callback('/#(['.self::HEX.']{6})|#(['.self::HEX.']{3})/i', array($this, 'hex2rgb'), $expression);
    469. 

  469.         $right_left = $last_left = 0;
    470. 

  470.         $rgb = $is_string = false;
    471. 

  471.         for ($i = 0; $i < strlen($expression); $i++)
    472. 

  472.         {
    473. 

  473.             $token = $expression{$i};
    474. 

  474.             if ($token == self::TOKEN_STRING && $is_string)
    475. 

  475.                 $is_string = false; else
    476. 

  476.             if ($token == self::TOKEN_STRING)
    477. 

  477.                 $is_string = true; else
    478. 

  478.             if ($token == self::TOKEN_LEFT_BRACKET)
    479. 

  479.             {
    480. 

  480.                 $last_left = $i;
    481. 

  481.                 $rgb = substr($expression, $i-3, 3) == 'rgb';
    482. 

  482.             } else
    483. 

  483.             if ($token == self::TOKEN_RIGHT_BRACKET)
    484. 

  484.             {
    485.
